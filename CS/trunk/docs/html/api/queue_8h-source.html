00029 // csRef<> is not threadsafe, and csPtr<> doesn't let us do anything with
00030 // the object referenced inside so this class, which is specifically designed
00031 // to communicate between threads, has no choice but to use raw pointers.
00032 // If this is used to communicate between threads, the 'feeder' thread should
00033 // incref the object before passing it into the queue.  The 'eater' thread
00034 // should NOT touch the refcount unless it's certain that no other thread will
00035 // be touching the refcount.  This makes cleanup ... interesting.
00036 // One possible method for cleanup is for the 'eater' thread which implicitely
00037 // holds a single reference (passed from the 'feeder') to wait for the
00038 // refcount to reach 1 before releasing it's refcount, since
00039 // a refcount of 1 means that it should have the only reference and thus
00040 // should be guaranteed to be the only thread working with the refcount.
00041 // Another possibility is for the 'eater' thread to queue this object back to
00042 // the 'feeder' thread, which will perform the decref itself.
00043 //
00044 // Note that if an object passed through this queue is meant to be accessed
00045 // from multiple threads at once it must of course contain threadsafe methods
00046 // itself.

00071 public:
00072   Queue() :
00073       head(0), tail(0), count(0), closed(false), dupecheck(false)

00086   void Clear()
00087   { 
00088     QEntry<T> *del;
00089     Lock(); 
00090     while (head)
00091     {
00092       del=head; 
00093       head=head->next;
00094       delete del;
00095     }
00096     tail=0;
00097 
00098     // Wake all waiting threads, queue is cleared
00099     queue_condition->Signal(true);
00100     Unlock();
00101   }

00103   int QueueEntry(T * data)
00104   {
00105     Lock();
00106 
00107     if (closed) return QUEUE_ERR_CLOSED;
00108 
00109     if (dupecheck && Find(data))
00110     {
00111       Unlock();
00112       return QUEUE_ERR_DUPE;
00113     }
00114 
00115     QEntry<T> *entry= new QEntry<T>();
00116     if (!entry)
00117     {
00118       Unlock();
00119       return QUEUE_ERR_NOMEM;
00120     }
00121     entry->data=data;
00122     entry->prev=tail;
00123     entry->next=0;
00124 
00125     if (!tail)
00126       head=entry;
00127     else
00128       tail->next=entry;
00129     tail=entry;
00130 
00131 
00132     // Signal one waiting thread to wake up
00133     queue_condition->Signal();
00134 
00135     Unlock();
00136     return QUEUE_SUCCESS;
00137   }
00138 
00139   // Note that even if wait==true, this function may return 0
00140   //  Such a situation is possible if:
00141   //  1) The Queue is cleared by destruction or a call to Clear()
00142   //  2) The condition wait is interrupted - possibly by signal arrival
00143   T * DequeueEntry(bool wait=false)
00144   {
00145     QEntry<T> *removed;
00146     T * ret=0;
00147 
00148     Lock();
00149 
00150     // Wait for an entry to be available if specified
00151     //  this is ignored if threading support is disabled
00152     //  since a single threaded application obviously cant
00153     //  add new entries to the queue while the program is stuck
00154     //  waiting.
00155     if (!head && wait)
00156       queue_condition->Wait(queue_mutex,0);
00157 
00158     // Remove the head entry from the queue
00159     if (head)
00160     {
00161       removed=head;
00162       head=head->next;
00163       if (head)
00164         head->prev=0;
00165       else
00166         tail=0;
00167       ret=removed->data;
00168       delete removed;
00169     }
00170     Unlock();
00171     return ret;
00172   }

00177   bool Find(T *data)
00178   {
00179     Lock();
00180     QEntry<T> *cur=head;
00181     while (cur)
00182     {
00183       if (((cur->data)) == (data))
00184       {
00185         Unlock();
00186         return true;
00187       }
00188       cur=cur->next;
00189     }
00190     Unlock();
00191     return false;
00192   }
