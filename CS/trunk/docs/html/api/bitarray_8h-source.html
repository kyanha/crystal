00059   {
00060     mNumBits = size;

00062     if (size == 0)

00075   inline void Trim()
00076   {
00077     unsigned extra_bits = mNumBits % cell_size;
00078     if (mLength > 0 && extra_bits != 0)
00079       mpStore[mLength - 1] &= ~((~(store_type) 0) << extra_bits);
00080   }
