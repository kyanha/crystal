00553   static int Compare (BitArray const& key1, BitArray const& key2)
00554   {
00555     csBitArrayStorageType const* p0 = key1.GetStore();
00556     csBitArrayStorageType const* p1 = key2.GetStore();
00557     size_t compareNum = MIN (key1.mLength, key2.mLength);
00558     size_t i = 0;
00559     for (; i < compareNum; i++)
00560       if (p0[i] != p1[i])
00561         return (int)p0[i] - (int)p1[i];
00562     if (key1.mLength > key2.mLength)
00563     {
00564       for (; i < key1.mLength; i++)
00565         if (p0[i] != 0)
00566           return (int)p0[i];
00567     }
00568     else
00569     {
00570       for (; i < key2.mLength; i++)
00571         if (p1[i] != 0)
00572           return -((int)p1[i]);
00573     }
00574     return 0;
00575   }
