00921   virtual void SetCapacityInternal (size_t NewSize, bool soft)
00922   {
00923     if (Data != 0) // If dynamic buffer already allocated, just re-use it.
00924       csStringBase::SetCapacityInternal(NewSize, soft);
00925     else
00926     {
00927       NewSize++; // Plus one for implicit null byte.
00928       if (NewSize <= LEN)
00929         miniused = NewSize;
00930       else
00931       {
00932         CS_ASSERT(MaxSize == 0);
00933         if (soft)
00934           NewSize = ComputeNewSize (NewSize);
00935         Data = new char[NewSize];
00936         MaxSize = NewSize;
00937         if (Size == 0)
00938           Data[0] = '\0';
00939         else
00940           memcpy(Data, minibuff, Size + 1);
00941       }
00942     }
00943   }
