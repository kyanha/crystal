00857 template<int LEN = 36>
00858 class csStringFast : public csStringBase
00859 {
00860 protected:
00861   char internalBuffer[LEN];
00862   virtual void SetCapacityInternal (size_t NewSize, bool extraSpace)
00863   {
00864     char* oldBuf = (MaxSize > LEN) ? Data : 0;
00865     char* newBuf;
00866     if (NewSize < LEN)
00867     {
00868       newBuf = internalBuffer;
00869       /* No need to honor extraSpace; later "reallocations" are free when
00870        * using the internal buffer, so no need to "allocate" more than
00871        * necessary. */
00872     }
00873     else
00874     {
00875       if (extraSpace)
00876         NewSize = ComputeNewSize (NewSize) + 1;
00877       newBuf = new char[NewSize];
00878     }
00879     if (newBuf != Data)
00880     {
00881       if (Data == 0 || Size == 0)
00882         newBuf[0] = '\0';
00883       else
00884         memcpy(newBuf, Data, Size + 1);
00885     }
00886 
00887     MaxSize = NewSize;
00888     Data = newBuf;
00889     delete[] oldBuf;
00890   }
