00911   virtual void SetCapacityInternal (size_t NewSize, bool soft)
00912   {
00913     if (Data != 0) // If dynamic buffer already allocated, just re-use it.
00914       csStringBase::SetCapacityInternal(NewSize, soft);
00915     else
00916     {
00917       NewSize++; // Plus one for implicit null byte.
00918       if (NewSize <= LEN)
00919         miniused = NewSize;
00920       else
00921       {
00922         CS_ASSERT(MaxSize == 0);
00923         if (soft)
00924           NewSize = ComputeNewSize (NewSize);
00925         Data = new char[NewSize];
00926         MaxSize = NewSize;
00927         if (Size == 0)
00928           Data[0] = '\0';
00929         else
00930           memcpy(Data, minibuff, Size + 1);
00931       }
00932     }
00933   }
