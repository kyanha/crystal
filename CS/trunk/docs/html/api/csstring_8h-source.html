00963   virtual void SetCapacityInternal (size_t NewSize, bool soft)
00964   {
00965     if (Data != 0) // If dynamic buffer already allocated, just re-use it.
00966       csStringBase::SetCapacityInternal(NewSize, soft);
00967     else
00968     {
00969       NewSize++; // Plus one for implicit null byte.
00970       if (NewSize <= LEN)
00971         miniused = NewSize;
00972       else
00973       {
00974         CS_ASSERT(MaxSize == 0);
00975         if (soft)
00976           NewSize = ComputeNewSize (NewSize);
00977         Data = new char[NewSize];
00978         MaxSize = NewSize;
00979         if (Size == 0)
00980           Data[0] = '\0';
00981         else
00982           memcpy(Data, minibuff, Size + 1);
00983       }
00984     }
00985   }
