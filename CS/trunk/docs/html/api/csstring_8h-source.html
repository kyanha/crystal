00924   virtual void SetCapacityInternal (size_t NewSize, bool soft)
00925   {
00926     if (Data != 0) // If dynamic buffer already allocated, just re-use it.
00927       csStringBase::SetCapacityInternal(NewSize, soft);
00928     else
00929     {
00930       NewSize++; // Plus one for implicit null byte.
00931       if (NewSize <= LEN)
00932         miniused = NewSize;
00933       else
00934       {
00935         CS_ASSERT(MaxSize == 0);
00936         if (soft)
00937           NewSize = ComputeNewSize (NewSize);
00938         Data = new char[NewSize];
00939         MaxSize = NewSize;
00940         if (Size == 0)
00941           Data[0] = '\0';
00942         else
00943           memcpy(Data, minibuff, Size + 1);
00944       }
00945     }
00946   }
