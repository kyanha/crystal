00923   virtual void SetCapacityInternal (size_t NewSize, bool soft)
00924   {
00925     if (Data != 0) // If dynamic buffer already allocated, just re-use it.
00926       csStringBase::SetCapacityInternal(NewSize, soft);
00927     else
00928     {
00929       NewSize++; // Plus one for implicit null byte.
00930       if (NewSize <= LEN)
00931         miniused = NewSize;
00932       else
00933       {
00934         CS_ASSERT(MaxSize == 0);
00935         if (soft)
00936           NewSize = ComputeNewSize (NewSize);
00937         Data = new char[NewSize];
00938         MaxSize = NewSize;
00939         if (Size == 0)
00940           Data[0] = '\0';
00941         else
00942           memcpy(Data, minibuff, Size + 1);
00943       }
00944     }
00945   }
