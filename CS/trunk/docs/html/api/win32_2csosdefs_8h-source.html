00442 // Microsoft Visual C++ compiler includes a very in-efficient 'memcpy'.
00443 // This also replaces the older 'better_memcpy', which was also not as
00444 // efficient as it could be ergo... heres a better solution.
00445 #if defined(CS_COMPILER_MSVC) && (_MSC_VER < 1300)
00446 #include <memory.h>
00447 #define memcpy fast_mem_copy
00448 static inline void* fast_mem_copy (void *dest, const void *src, int count)
00449 {
00450     __asm
00451     {
00452       mov               eax, count
00453       mov               esi, src
00454       mov               edi, dest
00455       xor               ecx, ecx
00456 
00457       // Check for 'short' moves
00458       cmp               eax, 16
00459       jl                do_short
00460                 
00461       // Move enough bytes to align 'dest'
00462       sub               ecx, edi
00463       and               ecx, 3
00464       je                skip
00465       sub               eax, ecx
00466       rep               movsb
00467 
00468       skip:
00469         mov             ecx, eax
00470         and             eax, 3
00471         shr             ecx, 2
00472         rep             movsd
00473         test    eax, eax
00474         je              end
00475 
00476       do_short:
00477         mov             ecx, eax
00478         rep             movsb
00479 
00480       end:
00481     }
00482 
00483     return dest;
00484 }
