00413 // Microsoft Visual C++ compiler includes a very in-efficient 'memcpy'.
00414 // This also replaces the older 'better_memcpy', which was also not as
00415 // efficient as it could be ergo... heres a better solution.
00416 #if defined(CS_COMPILER_MSVC) && (_MSC_VER < 1300)
00417 #include <memory.h>
00418 #define memcpy fast_mem_copy
00419 static inline void* fast_mem_copy (void *dest, const void *src, int count)
00420 {
00421     __asm
00422     {
00423       mov               eax, count
00424       mov               esi, src
00425       mov               edi, dest
00426       xor               ecx, ecx
00427 
00428       // Check for 'short' moves
00429       cmp               eax, 16
00430       jl                do_short
00431                 
00432       // Move enough bytes to align 'dest'
00433       sub               ecx, edi
00434       and               ecx, 3
00435       je                skip
00436       sub               eax, ecx
00437       rep               movsb
00438 
00439       skip:
00440         mov             ecx, eax
00441         and             eax, 3
00442         shr             ecx, 2
00443         rep             movsd
00444         test    eax, eax
00445         je              end
00446 
00447       do_short:
00448         mov             ecx, eax
00449         rep             movsb
00450 
00451       end:
00452     }
00453 
00454     return dest;
00455 }
00456 #endif
