00408 //======================================================================
00409 // For vertex based fog the following defines are used:
00410 #define CS_FOGTABLE_SIZE 64
00411 
00412 // Each texel in the fog table holds the fog alpha value at a certain
00413 // (distance*density).  The median distance parameter determines the
00414 // (distance*density) value represented by the texel at the center of
00415 // the fog table.  The fog calculation is:
00416 // alpha = 1.0 - exp( -(density*distance) / CS_FOGTABLE_MEDIANDISTANCE)
00417 #define CS_FOGTABLE_MEDIANDISTANCE 10.0f
00418 #define CS_FOGTABLE_MAXDISTANCE (CS_FOGTABLE_MEDIANDISTANCE * 2.0f)
00419 #define CS_FOGTABLE_DISTANCESCALE (1.0f / CS_FOGTABLE_MAXDISTANCE)
00420 
00421 // Fog (distance*density) is mapped to a texture coordinate and then
00422 // clamped.  This determines the clamp value.  Some drivers don't
00423 // like clamping textures so we must do it ourself
00424 #define CS_FOGTABLE_CLAMPVALUE 0.85f
00425 #define CS_FOG_MAXVALUE (CS_FOGTABLE_MAXDISTANCE * CS_FOGTABLE_CLAMPVALUE)
00426 //======================================================================
