00403 //======================================================================
00404 // For vertex based fog the following defines are used:
00405 #define CS_FOGTABLE_SIZE 64
00406 
00407 // Each texel in the fog table holds the fog alpha value at a certain
00408 // (distance*density).  The median distance parameter determines the
00409 // (distance*density) value represented by the texel at the center of
00410 // the fog table.  The fog calculation is:
00411 // alpha = 1.0 - exp( -(density*distance) / CS_FOGTABLE_MEDIANDISTANCE)
00412 #define CS_FOGTABLE_MEDIANDISTANCE 10.0f
00413 #define CS_FOGTABLE_MAXDISTANCE (CS_FOGTABLE_MEDIANDISTANCE * 2.0f)
00414 #define CS_FOGTABLE_DISTANCESCALE (1.0f / CS_FOGTABLE_MAXDISTANCE)
00415 
00416 // Fog (distance*density) is mapped to a texture coordinate and then
00417 // clamped.  This determines the clamp value.  Some drivers don't
00418 // like clamping textures so we must do it ourself
00419 #define CS_FOGTABLE_CLAMPVALUE 0.85f
00420 #define CS_FOG_MAXVALUE (CS_FOGTABLE_MAXDISTANCE * CS_FOGTABLE_CLAMPVALUE)
00421 //======================================================================
