00419 //======================================================================
00420 // For vertex based fog the following defines are used:
00421 #define CS_FOGTABLE_SIZE 64
00422 
00423 // Each texel in the fog table holds the fog alpha value at a certain
00424 // (distance*density).  The median distance parameter determines the
00425 // (distance*density) value represented by the texel at the center of
00426 // the fog table.  The fog calculation is:
00427 // alpha = 1.0 - exp( -(density*distance) / CS_FOGTABLE_MEDIANDISTANCE)
00428 #define CS_FOGTABLE_MEDIANDISTANCE 10.0f
00429 #define CS_FOGTABLE_MAXDISTANCE (CS_FOGTABLE_MEDIANDISTANCE * 2.0f)
00430 #define CS_FOGTABLE_DISTANCESCALE (1.0f / CS_FOGTABLE_MAXDISTANCE)
00431 
00432 // Fog (distance*density) is mapped to a texture coordinate and then
00433 // clamped.  This determines the clamp value.  Some drivers don't
00434 // like clamping textures so we must do it ourself
00435 #define CS_FOGTABLE_CLAMPVALUE 0.85f
00436 #define CS_FOG_MAXVALUE (CS_FOGTABLE_MAXDISTANCE * CS_FOGTABLE_CLAMPVALUE)
00437 //======================================================================
