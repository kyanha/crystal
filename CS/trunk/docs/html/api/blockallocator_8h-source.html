00096   {
00097     size_t i;
00098     for (i = 0 ; i < blocks.Length () ; i++)
00099     {

00216   {

00219     // This routine makes sure there is ALWAYS free space available.
00220     csBlock& freebl = blocks[firstfreeblock];

00248   void Free (T* el)
00249   {
00250     if (!el) return;
00251 
00252     size_t idx = FindBlock ((void*)el);
00253     CS_ASSERT (idx != (size_t)-1);
00254 
00255     el->~T();
00256 
00257 #ifdef CS_BLOCKALLOC_DEBUG
00258     memset (el, 0xfb, elsize);
00259 #endif
00260 
00261     // If the index is lower than the index of the first free block
00262     // then we update the firstfreeblock index.
00263     if (idx < firstfreeblock)
00264       firstfreeblock = idx;
00265 
00266     csBlock& bl = blocks[idx];
00267     if (bl.firstfree == 0)
00268     {
00269       // Block has no free items so we create the first free item
00270       // here.
00271       bl.firstfree = (csFreeList*)el;
00272       bl.firstfree->next = 0;
00273       bl.firstfree->numfree = 1;
00274     }
00275     else
00276     {
00277       csFreeList* p_el = (csFreeList*)el;
00278 
00279       if (p_el < bl.firstfree)
00280       {
00281         // New free element is before the current free element.
00282         if (((char*)bl.firstfree) - ((char*)p_el) == (int)elsize)
00283         {
00284           // New free element is directly before the current free
00285           // element.
00286           p_el->next = bl.firstfree->next;
00287           p_el->numfree = bl.firstfree->numfree+1;
00288         }
00289         else
00290         {
00291           // There is a gap.
00292           p_el->next = bl.firstfree;
00293           p_el->numfree = 1;
00294         }
00295 
00296         bl.firstfree = p_el;
00297       }
00298       else
00299       {
00300         // New free element is after the current free element.
00301         // First we find the two free blocks that enclose the new
00302         // free element.
00303         csFreeList* fl_before = bl.firstfree;
00304         csFreeList* fl_after = bl.firstfree->next;
00305         while (fl_after != 0 && fl_after < p_el)
00306         {
00307           fl_before = fl_after;
00308           fl_after = fl_after->next;
00309         }
00310 
00311         // 'fl_before_end' points to the memory right after the free block
00312         // that is directly before the new free element. We use
00313         // 'fl_before_end' later.
00314         char* fl_before_end = ((char*)fl_before) + fl_before->numfree * elsize;
00315 
00316         if (!fl_after)
00317         {
00318           // The new free element is after the last free block. First check
00319           // if the new free element directly follows that free block.
00320           // If so we can extend that.
00321           if (fl_before_end == (char*)p_el)
00322           {
00323             // Yes, extend.
00324             ++(fl_before->numfree);
00325           }
00326           else
00327           {
00328             // No, we have to create a new free block.
00329             p_el->next = 0;
00330             p_el->numfree = 1;
00331             fl_before->next = p_el;
00332           }
00333         }
00334         else
00335         {
00336           // We have a block before and after the new free block.
00337           // First we check if the new free item exactly between
00338           // fl_before and fl_after.
00339           if (fl_before_end == (char*)p_el
00340                 && ((char*)p_el) + elsize == (char*)fl_after)
00341           {
00342             // Perfect fit, merge fl_before and fl_after.
00343             fl_before->next = fl_after->next;
00344             fl_before->numfree = fl_before->numfree + 1 + fl_after->numfree;
00345           }
00346           else if (fl_before_end == (char*)p_el)
00347           {
00348             // New free item fits directly after fl_before.
00349             ++(fl_before->numfree);
00350           }
00351           else if (((char*)p_el) + elsize == (char*)fl_after)
00352           {
00353             // New free item fits directly before fl_after.
00354             fl_before->next = p_el;
00355             p_el->next = fl_after->next;
00356             p_el->numfree = fl_after->numfree+1;
00357           }
00358           else
00359           {
00360             // We need a new free block.
00361             fl_before->next = p_el;
00362             p_el->next = fl_after;
00363             p_el->numfree = 1;
00364           }
00365         }
00366       }
00367     }
