00095   size_t FindBlock (void* m)
00096   {
00097     size_t i;
00098     for (i = 0 ; i < blocks.Length () ; i++)
00099     {
00100       csBlock* b = &blocks[i];
00101       if (b->memory <= m)
00102       {
00103         char* eb = ((char*)b->memory) + blocksize;
00104         if (((char*)m) < eb)
00105           return i;
00106       }
00107     }

00198   void Compact ()
00199   {
00200     if (blocks.Length() == 0) return;
00201     size_t i = blocks.Length ()-1;
00202     while (i > firstfreeblock)

00214   T* Alloc ()
00215   {
00216     // This routine makes sure there is ALWAYS free space available.
00217     csBlock& freebl = blocks[firstfreeblock];
00218     void* ptr = (void*)(freebl.firstfree);
00219 
00220     if (freebl.firstfree->numfree >= 2)
00221     {
00222       // Still room in this block after allocation.
00223       csFreeList* nf = (csFreeList*)(((char*)ptr)+elsize);
00224       nf->next = freebl.firstfree->next;
00225       nf->numfree = freebl.firstfree->numfree-1;
00226       freebl.firstfree = nf;
00227     }
00228     else
00229     {
00230       // We need a new block later.
00231       freebl.firstfree = freebl.firstfree->next;
00232       if (!freebl.firstfree)
00233       {
00234         // This block has no more free space. We need a new one.
00235         FindAndUpdateFreeBlock ();
00236       }
00237     }

00245   void Free (T* el)
00246   {
00247     if (!el) return;
00248 
00249     size_t idx = FindBlock ((void*)el);
00250     CS_ASSERT (idx != (size_t)-1);
00251 
00252     el->~T();
00253 
00254 #ifdef CS_BLOCKALLOC_DEBUG
00255     memset (el, 0xfb, elsize);
00256 #endif
00257 
00258     // If the index is lower than the index of the first free block
00259     // then we update the firstfreeblock index.
00260     if (idx < firstfreeblock)
00261       firstfreeblock = idx;
00262 
00263     csBlock& bl = blocks[idx];
00264     if (bl.firstfree == 0)
00265     {
00266       // Block has no free items so we create the first free item
00267       // here.
00268       bl.firstfree = (csFreeList*)el;
00269       bl.firstfree->next = 0;
00270       bl.firstfree->numfree = 1;
00271     }
00272     else
00273     {
00274       csFreeList* p_el = (csFreeList*)el;
00275 
00276       if (p_el < bl.firstfree)
00277       {
00278         // New free element is before the current free element.
00279         if (((char*)bl.firstfree) - ((char*)p_el) == (int)elsize)
00280         {
00281           // New free element is directly before the current free
00282           // element.
00283           p_el->next = bl.firstfree->next;
00284           p_el->numfree = bl.firstfree->numfree+1;
00285         }
00286         else
00287         {
00288           // There is a gap.
00289           p_el->next = bl.firstfree;
00290           p_el->numfree = 1;
00291         }
00292 
00293         bl.firstfree = p_el;
00294       }
00295       else
00296       {
00297         // New free element is after the current free element.
00298         // First we find the two free blocks that enclose the new
00299         // free element.
00300         csFreeList* fl_before = bl.firstfree;
00301         csFreeList* fl_after = bl.firstfree->next;
00302         while (fl_after != 0 && fl_after < p_el)
00303         {
00304           fl_before = fl_after;
00305           fl_after = fl_after->next;
00306         }
00307 
00308         // 'fl_before_end' points to the memory right after the free block
00309         // that is directly before the new free element. We use
00310         // 'fl_before_end' later.
00311         char* fl_before_end = ((char*)fl_before) + fl_before->numfree * elsize;
00312 
00313         if (!fl_after)
00314         {
00315           // The new free element is after the last free block. First check
00316           // if the new free element directly follows that free block.
00317           // If so we can extend that.
00318           if (fl_before_end == (char*)p_el)
00319           {
00320             // Yes, extend.
00321             ++(fl_before->numfree);
00322           }
00323           else
00324           {
00325             // No, we have to create a new free block.
00326             p_el->next = 0;
00327             p_el->numfree = 1;
00328             fl_before->next = p_el;
00329           }
00330         }
00331         else
00332         {
00333           // We have a block before and after the new free block.
00334           // First we check if the new free item exactly between
00335           // fl_before and fl_after.
00336           if (fl_before_end == (char*)p_el
00337                 && ((char*)p_el) + elsize == (char*)fl_after)
00338           {
00339             // Perfect fit, merge fl_before and fl_after.
00340             fl_before->next = fl_after->next;
00341             fl_before->numfree = fl_before->numfree + 1 + fl_after->numfree;
00342           }
00343           else if (fl_before_end == (char*)p_el)
00344           {
00345             // New free item fits directly after fl_before.
00346             ++(fl_before->numfree);
00347           }
00348           else if (((char*)p_el) + elsize == (char*)fl_after)
00349           {
00350             // New free item fits directly before fl_after.
00351             fl_before->next = p_el;
00352             p_el->next = fl_after->next;
00353             p_el->numfree = fl_after->numfree+1;
00354           }
00355           else
00356           {
00357             // We need a new free block.
00358             fl_before->next = p_el;
00359             p_el->next = fl_after;
00360             p_el->numfree = 1;
00361           }
00362         }
00363       }
00364     }
