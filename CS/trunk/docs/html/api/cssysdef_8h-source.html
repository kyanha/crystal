00127 static inline bool isdir (const char *path, struct dirent *de)
00128 {
00129   int pathlen = strlen (path);
00130   char* fullname = new char[pathlen + 2 + strlen (de->d_name)];
00131   memcpy (fullname, path, pathlen + 1);
00132   if ((pathlen) && (fullname[pathlen-1] != CS_PATH_SEPARATOR))
00133   {
00134     fullname[pathlen++] = CS_PATH_SEPARATOR;
00135     fullname[pathlen] = 0;
00136   }
00137   strcat (&fullname [pathlen], de->d_name);
00138   struct stat st;
00139   stat (fullname, &st);
00140   delete[] fullname;
