00125 static inline bool isdir (const char *path, struct dirent *de)
00126 {
00127   int pathlen = strlen (path);
00128   char* fullname = new char[pathlen + 2 + strlen (de->d_name)];
00129   memcpy (fullname, path, pathlen + 1);
00130   if ((pathlen) && (fullname[pathlen-1] != CS_PATH_SEPARATOR))
00131   {
00132     fullname[pathlen++] = CS_PATH_SEPARATOR;
00133     fullname[pathlen] = 0;
00134   }
00135   strcat (&fullname [pathlen], de->d_name);
00136   struct stat st;
00137   stat (fullname, &st);
00138   delete[] fullname;
