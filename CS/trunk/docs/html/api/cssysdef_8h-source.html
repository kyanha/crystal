00113 static inline bool isdir (const char *path, struct dirent *de)
00114 {
00115   int pathlen = strlen (path);
00116   char* fullname = new char[pathlen + 2 + strlen (de->d_name)];
00117   memcpy (fullname, path, pathlen + 1);
00118   if ((pathlen) && (fullname[pathlen-1] != CS_PATH_SEPARATOR))
00119   {
00120     fullname[pathlen++] = CS_PATH_SEPARATOR;
00121     fullname[pathlen] = 0;
00122   }
00123   strcat (&fullname [pathlen], de->d_name);
00124   struct stat st;
00125   stat (fullname, &st);
00126   delete[] fullname;
00127   return ((st.st_mode & S_IFMT) == S_IFDIR);

00248   static int lastEntry = 0;                                            \
