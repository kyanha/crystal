00122   template <typename T>
00123   csEventError InternalRetrieveInt (const char* name, T& value) const
00124   {                                                             
00125     attribute* object = attributes.Get (name, 0);
00126     if (!object) return csEventErrNotFound;
00127     if ((object->type == csEventAttrInt) || (object->type == csEventAttrUInt))
00128     {                                                                   
00129       value = (T)object->intVal;
00130       const T rangeMin = (T)(1 << (sizeof(T) * 8 - 1));
00131       const T rangeMax = ~rangeMin;
00132       if ((object->intVal < rangeMin) || (object->intVal > rangeMax))
00133         return csEventErrLossy;
00134       else
00135         return csEventErrNone;
00136     }
00137     else
00138     {
00139       return InternalReportMismatch (object);
00140     }
00141   }
