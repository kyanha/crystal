00169 static inline float csQsqrt(float x)
00170 {
00171   float y0 = 0.0;
00172 
00173   if (x != 0.0)
00174   {
00175     float x0 = x * 0.5f;
00176 
00177     __asm__ __volatile__ ("frsqrte %0,%1" : "=f" (y0) : "f" (x));
00178     
00179     y0 = y0 * (1.5f - x0 * y0 * y0);
00180     y0 = (y0 * (1.5f - x0 * y0 * y0)) * x;
00181   };
00182     
00183   return y0;
00184 };
