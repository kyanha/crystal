00327 inline void csReporterHelper::ReportV(iObjectRegistry* reg, int severity,
00328   char const* msgId, char const* description, va_list args)
00329 {
00330   csRef<iReporter> reporter;
00331   if (reg && (reporter = CS_QUERY_REGISTRY (reg, iReporter)))
00332     reporter->ReportV (severity, msgId, description, args);
00333   else
00334   {
00335     /*
00336       @@@ The strncasecmp()s are there because sometimes reported messages
00337       start with "Warning", and a "Warning: Warning" output looks rather
00338       crappy. The correct fix is obviously to remove "Warning" prefixes
00339       when the reporter is used.
00340      */
00341     switch (severity)
00342     {
00343       case CS_REPORTER_SEVERITY_BUG:
00344         csPrintf(CS_ANSI_BK CS_ANSI_FM CS_ANSI_FI "BUG: " CS_ANSI_RST);
00345         break;
00346       case CS_REPORTER_SEVERITY_ERROR:
00347         if (strncasecmp (description, "error", 5) != 0)
00348           csPrintf(CS_ANSI_BK CS_ANSI_FR CS_ANSI_FI "ERROR: " CS_ANSI_RST);
00349         break;
00350       case CS_REPORTER_SEVERITY_WARNING:
00351         if (strncasecmp (description, "warning", 7) != 0)
00352           csPrintf(CS_ANSI_BK CS_ANSI_FY CS_ANSI_FI "WARNING: " CS_ANSI_RST);
00353         break;
00354       case CS_REPORTER_SEVERITY_NOTIFY:
00355         csPrintf ("NOTIFY: ");
00356         break;
00357       case CS_REPORTER_SEVERITY_DEBUG:
00358         csPrintf(CS_ANSI_BK CS_ANSI_FW CS_ANSI_FI "DEBUG: " CS_ANSI_RST);
00359         break;
00360     }
00361     csPrintfV(description, args);
00362     csPrintf("\n");
00363   }
00364 }
