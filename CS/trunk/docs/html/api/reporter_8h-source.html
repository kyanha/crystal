00330 inline void csReporterHelper::ReportV(iObjectRegistry* reg, int severity,
00331   char const* msgId, char const* description, va_list args)
00332 {
00333   csRef<iReporter> reporter;
00334   if (reg && (reporter = CS_QUERY_REGISTRY (reg, iReporter)))
00335     reporter->ReportV (severity, msgId, description, args);
00336   else
00337   {
00338     /*
00339       @@@ The csStrNCaseCmp()s are there because sometimes reported messages
00340       start with "Warning", and a "Warning: Warning" output looks rather
00341       crappy. The correct fix is obviously to remove "Warning" prefixes
00342       when the reporter is used.
00343      */
00344     switch (severity)
00345     {
00346       case CS_REPORTER_SEVERITY_BUG:
00347         csPrintf(CS_ANSI_BK CS_ANSI_FM CS_ANSI_FI "BUG: " CS_ANSI_RST);
00348         break;
00349       case CS_REPORTER_SEVERITY_ERROR:
00350         if (csStrNCaseCmp (description, "error", 5) != 0)
00351           csPrintf(CS_ANSI_BK CS_ANSI_FR CS_ANSI_FI "ERROR: " CS_ANSI_RST);
00352         break;
00353       case CS_REPORTER_SEVERITY_WARNING:
00354         if (csStrNCaseCmp (description, "warning", 7) != 0)
00355           csPrintf(CS_ANSI_BK CS_ANSI_FY CS_ANSI_FI "WARNING: " CS_ANSI_RST);
00356         break;
00357       case CS_REPORTER_SEVERITY_NOTIFY:
00358         csPrintf ("NOTIFY: ");
00359         break;
00360       case CS_REPORTER_SEVERITY_DEBUG:
00361         csPrintf(CS_ANSI_BK CS_ANSI_FW CS_ANSI_FI "DEBUG: " CS_ANSI_RST);
00362         break;
00363     }
00364     csPrintfV(description, args);
00365     csPrintf("\n");
00366   }
00367 }
