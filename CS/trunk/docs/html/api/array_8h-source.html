00594   size_t Push (T const& what)
00595   {
00596     if (((&what >= root) && (&what < root + Length())) && 
00597       (capacity < count + 1))
00598     {
00599       /*
00600         Special case: An element from this very array is pushed, and a 
00601         reallocation is needed. This could cause the passed ref to the
00602         element to be pushed to be read from deleted memory. Work
00603         around this.
00604        */
00605       size_t whatIndex = &what - root;
00606       SetLengthUnsafe (count + 1);
00607       ElementHandler::Construct (root + count - 1, root[whatIndex]);
00608     }
00609     else
00610     {
00611       SetLengthUnsafe (count + 1);
00612       ElementHandler::Construct (root + count - 1, what);
00613     }
00614     return count - 1;
00615   }
