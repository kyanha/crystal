00565   size_t Push (T const& what)
00566   {
00567     if (((&what >= root) && (&what < root + Length())) && 
00568       (capacity < count + 1))
00569     {
00570       /*
00571         Special case: An element from this very array is pushed, and a

00573         element to be pushed to be read from deleted memory. Work
00574         around this.

00579     }

00583       ElementHandler::Construct (root + count - 1, what);
00584     }
