00047 template<class Tpixel>
00048 class csPixMixerRGBA
00049 {
00050   uint32 rbMask, gaMask;
00051   int intAlphaInv;
00052   uint32 intRB, intGA;
00053   int gShift, gShift2;
00054   int alphaPrec;
00055 public:
00056   csPixMixerRGBA (csGraphics2D* G2D, Tpixel color, uint8 alpha)
00057   {
00058     alphaPrec = G2D->pfmt.GreenBits;
00059     const int alphaShift = 8 - alphaPrec;
00060     const int intAlpha = (alpha + 1) >> alphaShift;
00061     intAlphaInv = (255 - alpha + 1) >> alphaShift;
00062     gShift = G2D->pfmt.GreenShift;
00063     gShift2 = alphaPrec - gShift;
00064     // Makes the assumption that colors are laid out ARGB ...
00065     rbMask = G2D->pfmt.RedMask | G2D->pfmt.BlueMask;
00066     intRB = ((color & rbMask) * intAlpha) >> alphaPrec;
00067     gaMask = G2D->pfmt.GreenMask | G2D->pfmt.AlphaMask;
00068     intGA = (((color & gaMask) >> gShift) * intAlpha) >> gShift2;
00069   }
00070   inline void Mix (Tpixel& dest) const
00071   {
00072     const Tpixel destCol = dest;
00073     const Tpixel newRB = 
00074       ((intRB + (((destCol & rbMask) * intAlphaInv) >> alphaPrec)) & rbMask);
00075     const Tpixel newGA = 
00076       ((intGA + (((((destCol & gaMask) >> gShift) * intAlphaInv)) >> gShift2)) 
00077         & gaMask);
00078     dest = newRB | newGA;
00079   }
00080 };
