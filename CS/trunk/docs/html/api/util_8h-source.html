00178 {
00179   const unsigned int b[] = {0x2, 0xC, 0xF0, 0xFF00, 0xFFFF0000};
00180   const unsigned int S[] = {1, 2, 4, 8, 16};
00181   int i;
00182 
00183   register unsigned int c = 0; // result of log2(v) will go here
00184   for (i = 4; i >= 0; i--) // unroll for speed...
00185   {
00186     if (n & b[i])
00187     {
00188       n >>= S[i];

00190     }
