00095 #define CS_PROFSTART(tok,obj_reg) \
00096 static bool tok##__prof__init = false; \
00097 static uint32 tok##__prof__cnt = 0; \
00098 static uint32 tok##__prof__time = 0; \
00099 static uint32 tok##__prof__timemin = 1000000000; \
00100 static uint32 tok##__prof__timemax = 0; \
00101 if (!tok##__prof__init) \
00102 { \
00103   tok##__prof__init = true; \
00104   csRef<iProfiler> profiler = CS_QUERY_REGISTRY (obj_reg, iProfiler); \
00105   if (!profiler) \
00106   { \
00107     profiler.AttachNew (new csProfiler ()); \
00108     obj_reg->Register (profiler, "iProfiler"); \
00109   } \
00110   if (profiler) \
00111     profiler->RegisterProfilePoint (#tok,__FILE__, __LINE__, &tok##__prof__cnt, &tok##__prof__time, &tok##__prof__timemin, &tok##__prof__timemax); \
00112 } \
00113 uint32 tok##__prof__starttime; \
00114 CS_PROFTIME(tok##__prof__starttime)
00115 
00116 #define CS_PROFSTOP(tok) \
00117 { \
00118 uint32 prof__endtime; \
00119 CS_PROFTIME(prof__endtime); \
00120 uint32 prof__dt = prof__endtime - tok##__prof__starttime; \
00121 if (prof__dt < tok##__prof__timemin) tok##__prof__timemin = prof__dt; \
00122 if (prof__dt > tok##__prof__timemax) tok##__prof__timemax = prof__dt; \
00123 tok##__prof__time += prof__dt; \
00124 } \
00125 tok##__prof__cnt++
