00138 /*
00139     To be able to painlessly transfer files betwen platforms, we should
00140     avoid using native floating-point format. Here are a couple of routines
00141     that are guaranteed to work on all platforms.
00142 
00143     The floating point is converted to a fixed 1.7.25 bits format
00144     (one bit sign, 7 bits exponent, 25 bits mantissa) and back,
00145     so that we can binary store floating-point number without
00146     cross-platform problems. If you wonder why 1+7+25 = 33 while we
00147     only have 32 bits, we'll ommit the most significant bit of mantissa
00148     since it is always 1 (we use normalized numbers). This increases the
00149     precision twice.
00150 
00151     For double, we use one bit sign, 15 bits exponent, 49 bits mantissa.
00152 */
00153 
00155 static inline int32 float2long (float f)
00156 {
00157   int exp;
00158   int32 mant = csQroundSure (frexp (f, &exp) * 0x1000000);
00159   int32 sign = mant & 0x80000000;
00160   if (mant < 0) mant = -mant;
00161   if (exp > 63) exp = 63; else if (exp < -64) exp = -64;
00162   return sign | ((exp & 0x7f) << 24) | (mant & 0xffffff);
00163 }

00166 static inline float long2float (int32 l)
00167 {
00168   int exp = (l >> 24) & 0x7f;
00169   if (exp & 0x40) exp = exp | ~0x7f;
00170   float mant = float (l & 0x00ffffff) / 0x1000000;
00171   if (l & 0x80000000) mant = -mant;
00172   return (float) ldexp (mant, exp);
00173 }

00176 static inline int64 double2longlong (double d)
00177 {
00178   int exp;
00179   int64 mant = (int64) (frexp (d, &exp) * CONST_INT64(0x1000000000000));
00180   int64 sign = mant & CONST_INT64(0x800000000000000);
00181   if (mant < 0) mant = -mant;
00182   if (exp > 32767) exp = 32767; else if (exp < -32768) exp = -32768;
00183   return sign | ((int64 (exp) & 0x7fff) << 48) | (mant & CONST_INT64(0xffffffffffff));
00184 }

00187 static inline double longlong2double (int64 i)
00188 {
00189   int exp = (i >> 48) & 0x7fff;
00190   if (exp & 0x4000) exp = exp | ~0x7fff;
00191   double mant = double (i & CONST_INT64(0xffffffffffff)) / CONST_INT64(0x1000000000000);
00192   if (i & CONST_INT64(0x8000000000000000)) mant = -mant;
00193   return ldexp (mant, exp);
00194 }

00205 static inline short float2short (float f)
00206 {
00207   int exp;
00208   long mant = csQroundSure (frexp (f, &exp) * 0x1000);
00209   long sign = mant & 0x8000;
00210   if (mant < 0) mant = -mant;
00211   if (exp > 7) mant = 0x7ff, exp = 7; else if (exp < -8) mant = 0, exp = -8;
00212   return short(sign | ((exp & 0xf) << 11) | (mant & 0x7ff));
00213 }

00216 static inline float short2float (short s)
00217 {
00218   int exp = (s >> 11) & 0xf;
00219   if (exp & 0x8) exp = exp | ~0xf;
00220   float mant = float ((s & 0x07ff) | 0x0800) / 0x1000;
00221   if (s & 0x8000) mant = -mant;
00222   return (float) ldexp (mant, exp);
