00327 /*
00328     To be able to painlessly transfer files betwen platforms, we should
00329     avoid using native floating-point format. Here are a couple of routines
00330     that are guaranteed to work on all platforms.
00331 
00332     The floating point is converted to a fixed 1.7.25 bits format
00333     (one bit sign, 7 bits exponent, 25 bits mantissa) and back,
00334     so that we can binary store floating-point number without
00335     cross-platform problems. If you wonder why 1+7+25 = 33 while we
00336     only have 32 bits, we'll ommit the most significant bit of mantissa
00337     since it is always 1 (we use normalized numbers). This increases the
00338     precision twice.
00339 
00340     For double, we use one bit sign, 15 bits exponent, 49 bits mantissa.
00341 */
00342 
00347 /*CS_DEPRECATED_METHOD*/ static inline int32 csFloatToLong (float f)
00348 {
00349   int exp;
00350   int32 mant = csQroundSure (frexp (f, &exp) * 0x1000000);
00351   int32 sign = mant & 0x80000000;
00352   if (mant < 0) mant = -mant;
00353   if (exp > 63) exp = 63; else if (exp < -64) exp = -64;
00354   return sign | ((exp & 0x7f) << 24) | (mant & 0xffffff);
00355 }

00361 /*CS_DEPRECATED_METHOD*/ static inline float csLongToFloat (int32 l)
00362 {
00363   int exp = (l >> 24) & 0x7f;
00364   if (exp & 0x40) exp = exp | ~0x7f;
00365   float mant = float (l & 0x00ffffff) / 0x1000000;
00366   if (l & 0x80000000) mant = -mant;
00367   return (float) ldexp (mant, exp);
00368 }

00370 /* Implementation note: csDoubleToLongLong() and csLongLongToDouble()
00371  *
00372  * We avoid use of CONST_INT64() because 64-bit constants are illegal with g++
00373  * under -ansi -pedantic, and we want this header to be useful to external
00374  * projects which use -ansi -pedantic.  Instead, we use bit shifts, such as (1
00375  * << 59), and construct `mask' manually.
00376  */
00377 
00382 /*CS_DEPRECATED_METHOD*/ static inline int64 csDoubleToLongLong (double d)
00383 {
00384   int exp;
00385   int64 mant = (int64) (frexp (d, &exp) * ((int64)1 << 48));
00386   int64 sign = mant & ((int64)1 << 59);
00387   if (mant < 0) mant = -mant;
00388   if (exp > 32767) exp = 32767; else if (exp < -32768) exp = -32768;
00389   int64 const mask = ((uint64)0xffff << 32) | (uint64)0xffffffff;
00390   return sign | ((int64 (exp) & 0x7fff) << 48) | (mant & mask);
00391 }

00397 /*CS_DEPRECATED_METHOD*/ static inline double csLongLongToDouble (int64 i)
00398 {
00399   int exp = (i >> 48) & 0x7fff;
00400   if (exp & 0x4000) exp = exp | ~0x7fff;
00401   int64 const mask = ((uint64)0xffff << 32) | (uint64)0xffffffff;
00402   double mant = double (i & mask) / ((int64)1 << 48);
00403   if (i & ((int64)1 << 59)) mant = -mant;
00404   return ldexp (mant, exp);
00405 }

00407 /* *\name Floating point conversions
00408  * These routines are used for converting floating-point numbers
00409  * into 16-bit shorts and back. This is useful for low-precision data.
00410  * They use the 1.4.12 format. The range of numbers that can be represented
00411  * in this format is from 2^-8 to 2^7. The precision for numbers near to
00412  * 2^-8 (0.00390625) is near 0.000001, for numbers near 2^7 (128) is near 0.03.
00413  * @{ */
00414 
00419 /*CS_DEPRECATED_METHOD*/ static inline short csFloatToShort (float f)
00420 {
00421   int exp;
00422   long mant = csQroundSure (frexp (f, &exp) * 0x1000);
00423   long sign = mant & 0x8000;
00424   if (mant < 0) mant = -mant;
00425   if (exp > 7) mant = 0x7ff, exp = 7; else if (exp < -8) mant = 0, exp = -8;
00426   return short(sign | ((exp & 0xf) << 11) | (mant & 0x7ff));
00427 }

00433 /*CS_DEPRECATED_METHOD*/ static inline float csShortToFloat (short s)
00434 {
00435   int exp = (s >> 11) & 0xf;
00436   if (exp & 0x8) exp = exp | ~0xf;
00437   float mant = float ((s & 0x07ff) | 0x0800) / 0x1000;
00438   if (s & 0x8000) mant = -mant;
00439   return (float) ldexp (mant, exp);
00440 }
