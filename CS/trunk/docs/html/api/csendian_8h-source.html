00300 /*
00301     To be able to painlessly transfer files betwen platforms, we should
00302     avoid using native floating-point format. Here are a couple of routines
00303     that are guaranteed to work on all platforms.
00304 
00305     The floating point is converted to a fixed 1.7.25 bits format
00306     (one bit sign, 7 bits exponent, 25 bits mantissa) and back,
00307     so that we can binary store floating-point number without
00308     cross-platform problems. If you wonder why 1+7+25 = 33 while we
00309     only have 32 bits, we'll ommit the most significant bit of mantissa
00310     since it is always 1 (we use normalized numbers). This increases the
00311     precision twice.
00312 
00313     For double, we use one bit sign, 15 bits exponent, 49 bits mantissa.
00314 */
00315 
00320 static inline int32 csFloatToLong (float f)
00321 {
00322   int exp;
00323   int32 mant = csQroundSure (frexp (f, &exp) * 0x1000000);
00324   int32 sign = mant & 0x80000000;
00325   if (mant < 0) mant = -mant;
00326   if (exp > 63) exp = 63; else if (exp < -64) exp = -64;
00327   return sign | ((exp & 0x7f) << 24) | (mant & 0xffffff);
00328 }

00331 static inline float csLongToFloat (int32 l)
00332 {
00333   int exp = (l >> 24) & 0x7f;
00334   if (exp & 0x40) exp = exp | ~0x7f;
00335   float mant = float (l & 0x00ffffff) / 0x1000000;
00336   if (l & 0x80000000) mant = -mant;
00337   return (float) ldexp (mant, exp);
00338 }

00340 /* Implementation note: csDoubleToLongLong() and csLongLongToDouble()
00341  *
00342  * We avoid use of CONST_INT64() because 64-bit constants are illegal with g++
00343  * under -ansi -pedantic, and we want this header to be useful to external
00344  * projects which use -ansi -pedantic.  Instead, we use bit shifts, such as (1
00345  * << 59), and construct `mask' manually.
00346  */
00347 
00349 static inline int64 csDoubleToLongLong (double d)
00350 {
00351   int exp;
00352   int64 mant = (int64) (frexp (d, &exp) * ((int64)1 << 48));
00353   int64 sign = mant & ((int64)1 << 59);
00354   if (mant < 0) mant = -mant;
00355   if (exp > 32767) exp = 32767; else if (exp < -32768) exp = -32768;
00356   int64 const mask = ((uint64)0xffff << 32) | (uint64)0xffffffff;
00357   return sign | ((int64 (exp) & 0x7fff) << 48) | (mant & mask);
00358 }

00361 static inline double csLongLongToDouble (int64 i)
00362 {
00363   int exp = (i >> 48) & 0x7fff;
00364   if (exp & 0x4000) exp = exp | ~0x7fff;
00365   int64 const mask = ((uint64)0xffff << 32) | (uint64)0xffffffff;
00366   double mant = double (i & mask) / ((int64)1 << 48);
00367   if (i & ((int64)1 << 59)) mant = -mant;
00368   return ldexp (mant, exp);
00369 }

00371 /* *\name Floating point conversions
00372  * The following routines are used for converting floating-point numbers
00373  * into 16-bit shorts and back. This is useful for low-precision data.
00374  * They use the 1.4.12 format. The range of numbers that can be represented
00375  * in this format is from 2^-8 to 2^7. The precision for numbers near to
00376  * 2^-8 (0.00390625) is near 0.000001, for numbers near 2^7 (128) is near 0.03.
00377  * @{ */
00378 
00380 static inline short csFloatToShort (float f)
00381 {
00382   int exp;
00383   long mant = csQroundSure (frexp (f, &exp) * 0x1000);
00384   long sign = mant & 0x8000;
00385   if (mant < 0) mant = -mant;
00386   if (exp > 7) mant = 0x7ff, exp = 7; else if (exp < -8) mant = 0, exp = -8;
00387   return short(sign | ((exp & 0xf) << 11) | (mant & 0x7ff));
00388 }

00391 static inline float csShortToFloat (short s)
00392 {
00393   int exp = (s >> 11) & 0xf;
00394   if (exp & 0x8) exp = exp | ~0xf;
00395   float mant = float ((s & 0x07ff) | 0x0800) / 0x1000;
00396   if (s & 0x8000) mant = -mant;
00397   return (float) ldexp (mant, exp);
00398 }
