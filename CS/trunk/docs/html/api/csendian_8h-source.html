00342 /*
00343     To be able to painlessly transfer files betwen platforms, we should
00344     avoid using native floating-point format. Here are a couple of routines
00345     that are guaranteed to work on all platforms.
00346 
00347     The floating point is converted to a fixed 1.7.25 bits format
00348     (one bit sign, 7 bits exponent, 25 bits mantissa) and back,
00349     so that we can binary store floating-point number without
00350     cross-platform problems. If you wonder why 1+7+25 = 33 while we
00351     only have 32 bits, we'll ommit the most significant bit of mantissa
00352     since it is always 1 (we use normalized numbers). This increases the
00353     precision twice.
00354 
00355     For double, we use one bit sign, 15 bits exponent, 49 bits mantissa.
00356 */
00357 
00362 /*CS_DEPRECATED_METHOD*/ static inline int32 csFloatToLong (float f)
00363 {
00364   int exp;
00365   int32 mant = csQroundSure (frexp (f, &exp) * 0x1000000);
00366   int32 sign = mant & 0x80000000;
00367   if (mant < 0) mant = -mant;
00368   if (exp > 63) exp = 63; else if (exp < -64) exp = -64;
00369   return sign | ((exp & 0x7f) << 24) | (mant & 0xffffff);
00370 }

00376 /*CS_DEPRECATED_METHOD*/ static inline float csLongToFloat (int32 l)
00377 {
00378   int exp = (l >> 24) & 0x7f;
00379   if (exp & 0x40) exp = exp | ~0x7f;
00380   float mant = float (l & 0x00ffffff) / 0x1000000;
00381   if (l & 0x80000000) mant = -mant;
00382   return (float) ldexp (mant, exp);
00383 }

00385 /* Implementation note: csDoubleToLongLong() and csLongLongToDouble()
00386  *
00387  * We avoid use of CONST_INT64() because 64-bit constants are illegal with g++
00388  * under -ansi -pedantic, and we want this header to be useful to external
00389  * projects which use -ansi -pedantic.  Instead, we use bit shifts, such as (1
00390  * << 59), and construct `mask' manually.
00391  */

00397 /*CS_DEPRECATED_METHOD*/ static inline int64 csDoubleToLongLong (double d)
00398 {
00399   int exp;
00400   int64 mant = (int64) (frexp (d, &exp) * ((int64)1 << 48));
00401   int64 sign = mant & ((int64)1 << 59);
00402   if (mant < 0) mant = -mant;
00403   if (exp > 32767) exp = 32767; else if (exp < -32768) exp = -32768;
00404   int64 const mask = ((uint64)0xffff << 32) | (uint64)0xffffffff;
00405   return sign | ((int64 (exp) & 0x7fff) << 48) | (mant & mask);
00406 }

00412 /*CS_DEPRECATED_METHOD*/ static inline double csLongLongToDouble (int64 i)
00413 {
00414   int exp = (i >> 48) & 0x7fff;
00415   if (exp & 0x4000) exp = exp | ~0x7fff;
00416   int64 const mask = ((uint64)0xffff << 32) | (uint64)0xffffffff;
00417   double mant = double (i & mask) / ((int64)1 << 48);
00418   if (i & ((int64)1 << 59)) mant = -mant;
00419   return ldexp (mant, exp);
00420 }

00422 /* *\name Floating point conversions
00423  * These routines are used for converting floating-point numbers
00424  * into 16-bit shorts and back. This is useful for low-precision data.
00425  * They use the 1.4.12 format. The range of numbers that can be represented
00426  * in this format is from 2^-8 to 2^7. The precision for numbers near to
00427  * 2^-8 (0.00390625) is near 0.000001, for numbers near 2^7 (128) is near 0.03.
00428  * @{ */
00429 
00434 /*CS_DEPRECATED_METHOD*/ static inline short csFloatToShort (float f)
00435 {
00436   int exp;
00437   long mant = csQroundSure (frexp (f, &exp) * 0x1000);
00438   long sign = mant & 0x8000;
00439   if (mant < 0) mant = -mant;
00440   if (exp > 7) mant = 0x7ff, exp = 7; else if (exp < -8) mant = 0, exp = -8;
00441   return short(sign | ((exp & 0xf) << 11) | (mant & 0x7ff));
00442 }

00448 /*CS_DEPRECATED_METHOD*/ static inline float csShortToFloat (short s)
00449 {
00450   int exp = (s >> 11) & 0xf;
00451   if (exp & 0x8) exp = exp | ~0xf;
00452   float mant = float ((s & 0x07ff) | 0x0800) / 0x1000;
00453   if (s & 0x8000) mant = -mant;
00454   return (float) ldexp (mant, exp);
00455 }
