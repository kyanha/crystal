00099 // A definition for a line operation.
00100 struct csLineOperation
00101 {
00102   uint8 op;             // One of OP_...
00103   // All coordinates are with 0,0 relative to top-left of tile.
00104   // x coordinates are also shifted 16 to the left.
00105   int x1;               // Start of line.
00106   int y1;               // Start of line. Not used with OP_FULLVLINE.
00107   int x2;               // End of line. Only used with OP_LINE.
00108   int y2;               // End of line. Not used with OP_FULLVLINE.
00109   int dx;               // Slope to add to x1 (shifted 16 to left).
00110 };
