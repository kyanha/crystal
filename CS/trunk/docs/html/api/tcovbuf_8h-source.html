00089 // A definition for a line operation.
00090 struct csLineOperation
00091 {
00092   uint8 op;             // One of OP_...
00093   // All coordinates are with 0,0 relative to top-left of tile.
00094   // x coordinates are also shifted 16 to the left.
00095   int x1;               // Start of line.
00096   int y1;               // Start of line. Not used with OP_FULLVLINE.
00097   int x2;               // End of line. Only used with OP_LINE.
00098   int y2;               // End of line. Not used with OP_FULLVLINE.
00099   int dx;               // Slope to add to x1 (shifted 16 to left).
00100 };
