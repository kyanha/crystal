00027 class csG2DDrawLine
00028 {
00029 public:
00030   static void DrawLine (csGraphics2D* G2D, float x1, float y1, float x2, float y2,
00031     Tpixel color, uint8 alpha)
00032   {
00033     Tpixmixer mixer (G2D, color, alpha);
00034     unsigned char* Memory = G2D->Memory;
00035     const int* LineAddress = G2D->LineAddress;
00036 
00037     int fx1 = csQint (x1), fx2 = csQint (x2),
00038         fy1 = csQint (y1), fy2 = csQint (y2);
00039     
00040     if (fy1 == fy2)
00041     {
00042       if (fx2 - fx1)
00043       {
00044         if (fx1 > fx2) { int tmp = fx1; fx1 = fx2; fx2 = tmp; }
00045         int count = fx2 - fx1 + 1;
00046         register Tpixel* dest = (Tpixel*)G2D->GetPixelAt (fx1, fy1);
00047         while (count--) mixer.Mix (*dest++);
00048       }
00049       else
00050         mixer.Mix (*((Tpixel*)G2D->GetPixelAt (fx1, fy1)));
00051     }
00052     else if (abs (fx2 - fx1) > abs (fy2 - fy1))
00053     {
00054       // Transform floating-point format to 16.16 fixed-point
00055       fy1 = csQint16 (y1); fy2 = csQint16 (y2);
00056   
00057       if (fx1 > fx2)
00058       {
00059         int tmp = fx1; fx1 = fx2; fx2 = tmp;
00060         tmp = fy1; fy1 = fy2; fy2 = tmp;
00061       }
00062   
00063       // delta Y can be negative
00064       int deltay = (fy2 - fy1) / (fx2 - fx1 + 1);
00065   
00066       {                                                         
00067         int x, y;                                               
00068         for (x = fx1, y = fy1 + deltay / 2; x <= fx2; x++)      
00069         {                                                               
00070           Tpixel* p = (Tpixel*)(Memory +                        
00071             (x * sizeof (Tpixel) + LineAddress [y >> 16]));     
00072           mixer.Mix (*p); y += deltay;                                  
00073         }                                                               
00074       }
00075     }
00076     else
00077     {
00078       // Transform floating-point format to 16.16 fixed-point
00079       fx1 = csQint16 (x1); fx2 = csQint16 (x2);
00080   
00081       if (fy1 > fy2)
00082       {
00083         int tmp = fy1; fy1 = fy2; fy2 = tmp;
00084         tmp = fx1; fx1 = fx2; fx2 = tmp;
00085       }
00086   
00087       // delta X can be negative
00088       int deltax = (fx2 - fx1) / (fy2 - fy1 + 1);
00089   
00090       {                             
00091         int x, y; 
00092         for (x = fx1 + deltax / 2, y = fy1; y <= fy2; y++)  
00093         {                               
00094           Tpixel* p = (Tpixel*)(Memory +             
00095             ((x >> 16) * sizeof (Tpixel) + LineAddress [y]));  
00096           mixer.Mix (*p); x += deltax;                  
00097         }                               
00098       }
00099     }
