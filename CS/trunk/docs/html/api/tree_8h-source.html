00110     {

00112       if (TreeFunc (node, param, stopOnSuccess))
00113         foundNode = node;
00114       if (!node->IsLeaf () && (SelBranch==0 || SelBranch (node)))
00115       {
00116         size_t i;
00117         for (i=0; i < node->children.Length (); i++ )

00120     }
