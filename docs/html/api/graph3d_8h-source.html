00421 //======================================================================
00422 // For vertex based fog the following defines are used:
00423 #define CS_FOGTABLE_SIZE 64
00424 
00425 // Each texel in the fog table holds the fog alpha value at a certain
00426 // (distance*density).  The median distance parameter determines the
00427 // (distance*density) value represented by the texel at the center of
00428 // the fog table.  The fog calculation is:
00429 // alpha = 1.0 - exp( -(density*distance) / CS_FOGTABLE_MEDIANDISTANCE)
00430 #define CS_FOGTABLE_MEDIANDISTANCE 10.0f
00431 #define CS_FOGTABLE_MAXDISTANCE (CS_FOGTABLE_MEDIANDISTANCE * 2.0f)
00432 #define CS_FOGTABLE_DISTANCESCALE (1.0f / CS_FOGTABLE_MAXDISTANCE)
00433 
00434 // Fog (distance*density) is mapped to a texture coordinate and then
00435 // clamped.  This determines the clamp value.  Some drivers don't
00436 // like clamping textures so we must do it ourself
00437 #define CS_FOGTABLE_CLAMPVALUE 0.85f
00438 #define CS_FOG_MAXVALUE (CS_FOGTABLE_MAXDISTANCE * CS_FOGTABLE_CLAMPVALUE)
00439 //======================================================================
