00402 //======================================================================
00403 // For vertex based fog the following defines are used:
00404 #define CS_FOGTABLE_SIZE 64
00405 
00406 // Each texel in the fog table holds the fog alpha value at a certain
00407 // (distance*density).  The median distance parameter determines the
00408 // (distance*density) value represented by the texel at the center of
00409 // the fog table.  The fog calculation is:
00410 // alpha = 1.0 - exp( -(density*distance) / CS_FOGTABLE_MEDIANDISTANCE)
00411 #define CS_FOGTABLE_MEDIANDISTANCE 10.0f
00412 #define CS_FOGTABLE_MAXDISTANCE (CS_FOGTABLE_MEDIANDISTANCE * 2.0f)
00413 #define CS_FOGTABLE_DISTANCESCALE (1.0f / CS_FOGTABLE_MAXDISTANCE)
00414 
00415 // Fog (distance*density) is mapped to a texture coordinate and then
00416 // clamped.  This determines the clamp value.  Some drivers don't
00417 // like clamping textures so we must do it ourself
00418 #define CS_FOGTABLE_CLAMPVALUE 0.85f
00419 #define CS_FOG_MAXVALUE (CS_FOGTABLE_MAXDISTANCE * CS_FOGTABLE_CLAMPVALUE)
