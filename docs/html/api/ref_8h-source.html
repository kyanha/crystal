00224   csRef& operator = (T* newobj)
00225   {
00226     if (obj != newobj)
00227     {
00228       T* oldobj = obj;
00229       // It is very important to first assign the new value to
00230       // 'obj' BEFORE calling DecRef() on the old object. Otherwise
