00025 template<class Tpixel, class Tpixmixer>
00026 class csG2DDrawLine
00027 {
00028 public:
00029   static void DrawLine (csGraphics2D* G2D, float x1, float y1, float x2, float y2,
00030     Tpixel color, uint8 alpha)
00031   {
00032     Tpixmixer mixer (G2D, color, alpha);
00033     unsigned char* Memory = G2D->Memory;
00034     const int* LineAddress = G2D->LineAddress;
00035 
00036     int fx1 = csQint (x1), fx2 = csQint (x2),
00037         fy1 = csQint (y1), fy2 = csQint (y2);
00038     
00039     if (fy1 == fy2)
00040     {
00041       if (fx2 - fx1)
00042       {
00043         if (fx1 > fx2) { int tmp = fx1; fx1 = fx2; fx2 = tmp; }
00044         int count = fx2 - fx1 + 1;
00045         register Tpixel* dest = (Tpixel*)G2D->GetPixelAt (fx1, fy1);
00046         while (count--) mixer.Mix (*dest++);
00047       }
00048       else
00049         mixer.Mix (*((Tpixel*)G2D->GetPixelAt (fx1, fy1)));
00050     }
00051     else if (abs (fx2 - fx1) > abs (fy2 - fy1))
00052     {
00053       // Transform floating-point format to 16.16 fixed-point
00054       fy1 = csQint16 (y1); fy2 = csQint16 (y2);
00055   
00056       if (fx1 > fx2)
00057       {
00058         int tmp = fx1; fx1 = fx2; fx2 = tmp;
00059         tmp = fy1; fy1 = fy2; fy2 = tmp;
00060       }
00061   
00062       // delta Y can be negative
00063       int deltay = (fy2 - fy1) / (fx2 - fx1 + 1);
00064   
00065       {                                                         
00066         int x, y;                                               
00067         for (x = fx1, y = fy1 + deltay / 2; x <= fx2; x++)      
00068         {                                                               
00069           Tpixel* p = (Tpixel*)(Memory +                        
00070             (x * sizeof (Tpixel) + LineAddress [y >> 16]));     
00071           mixer.Mix (*p); y += deltay;                                  
00072         }                                                               
00073       }
00074     }
00075     else
00076     {
00077       // Transform floating-point format to 16.16 fixed-point
00078       fx1 = csQint16 (x1); fx2 = csQint16 (x2);
00079   
00080       if (fy1 > fy2)
00081       {
00082         int tmp = fy1; fy1 = fy2; fy2 = tmp;
00083         tmp = fx1; fx1 = fx2; fx2 = tmp;
00084       }
00085   
00086       // delta X can be negative
00087       int deltax = (fx2 - fx1) / (fy2 - fy1 + 1);
00088   
00089       {                             
00090         int x, y; 
00091         for (x = fx1 + deltax / 2, y = fy1; y <= fy2; y++)  
00092         {                               
00093           Tpixel* p = (Tpixel*)(Memory +             
00094             ((x >> 16) * sizeof (Tpixel) + LineAddress [y]));  
00095           mixer.Mix (*p); x += deltax;                  
00096         }                               
00097       }
00098     }
00099     
00100   }
00101 };
