00447 // Microsoft Visual C++ compiler includes a very in-efficient 'memcpy'.
00448 // This also replaces the older 'better_memcpy', which was also not as
00449 // efficient as it could be ergo... heres a better solution.
00450 #if defined(COMP_VC) && (_MSC_VER < 1300)
00451 #include <memory.h>
00452 #define memcpy fast_mem_copy
00453 static inline void* fast_mem_copy (void *dest, const void *src, int count)
00454 {
00455     __asm
00456     {
00457       mov               eax, count
00458       mov               esi, src
00459       mov               edi, dest
00460       xor               ecx, ecx
00461 
00462       // Check for 'short' moves
00463       cmp               eax, 16
00464       jl                do_short
00465                 
00466       // Move enough bytes to align 'dest'
00467       sub               ecx, edi
00468       and               ecx, 3
00469       je                skip
00470       sub               eax, ecx
00471       rep               movsb
00472 
00473       skip:
00474         mov             ecx, eax
00475         and             eax, 3
00476         shr             ecx, 2
00477         rep             movsd
00478         test    eax, eax
00479         je              end
00480 
00481       do_short:
00482         mov             ecx, eax
00483         rep             movsb
00484 
00485       end:
00486     }
00487 
00488     return dest;
00489 }
