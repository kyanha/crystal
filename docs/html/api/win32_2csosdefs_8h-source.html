00401 // Microsoft Visual C++ compiler includes a very in-efficient 'memcpy'.
00402 // This also replaces the older 'better_memcpy', which was also not as
00403 // efficient as it could be ergo... heres a better solution.
00404 #if defined(CS_COMPILER_MSVC) && (_MSC_VER < 1300)
00405 #include <memory.h>
00406 #define memcpy fast_mem_copy
00407 static inline void* fast_mem_copy (void *dest, const void *src, int count)
00408 {
00409     __asm
00410     {
00411       mov               eax, count
00412       mov               esi, src
00413       mov               edi, dest
00414       xor               ecx, ecx
00415 
00416       // Check for 'short' moves
00417       cmp               eax, 16
00418       jl                do_short
00419                 
00420       // Move enough bytes to align 'dest'
00421       sub               ecx, edi
00422       and               ecx, 3
00423       je                skip
00424       sub               eax, ecx
00425       rep               movsb
00426 
00427       skip:
00428         mov             ecx, eax
00429         and             eax, 3
00430         shr             ecx, 2
00431         rep             movsd
00432         test    eax, eax
00433         je              end
00434 
00435       do_short:
00436         mov             ecx, eax
00437         rep             movsb
00438 
00439       end:
00440     }
00441 
00442     return dest;
00443 }
00444 #endif
