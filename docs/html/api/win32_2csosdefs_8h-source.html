00396 // Microsoft Visual C++ compiler includes a very in-efficient 'memcpy'.
00397 // This also replaces the older 'better_memcpy', which was also not as
00398 // efficient as it could be ergo... heres a better solution.
00399 #if defined(CS_COMPILER_MSVC) && (_MSC_VER < 1300)
00400 #include <memory.h>
00401 #define memcpy fast_mem_copy
00402 static inline void* fast_mem_copy (void *dest, const void *src, int count)
00403 {
00404     __asm
00405     {
00406       mov               eax, count
00407       mov               esi, src
00408       mov               edi, dest
00409       xor               ecx, ecx
00410 
00411       // Check for 'short' moves
00412       cmp               eax, 16
00413       jl                do_short
00414                 
00415       // Move enough bytes to align 'dest'
00416       sub               ecx, edi
00417       and               ecx, 3
00418       je                skip
00419       sub               eax, ecx
00420       rep               movsb
00421 
00422       skip:
00423         mov             ecx, eax
00424         and             eax, 3
00425         shr             ecx, 2
00426         rep             movsd
00427         test    eax, eax
00428         je              end
00429 
00430       do_short:
00431         mov             ecx, eax
00432         rep             movsb
00433 
00434       end:
00435     }
00436 
00437     return dest;
00438 }
00439 #endif
