00210   for (size_t i = 0 ; i < scfWeakRefOwners->Length () ; i++)            \
00211   {                                                                     \
00212     iBase** p = (*scfWeakRefOwners)[i];                                 \

00215   delete scfWeakRefOwners;                                              \
00216   scfWeakRefOwners = 0;                                                 \
00217 }
