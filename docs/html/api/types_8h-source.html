00070         const size_t floatNum, size_t sv, size_t fv, int sy)
00071       {
00072         const csVector3 vsv (vertices[sv]);
00073         const csVector3 vfv (vertices[fv]);
00074   
00075         float dy = vsv.y - vfv.y;
00076         if (dy)
00077         {
00078           float inv_dy = 1 / dy;
00079           x = vsv.x;
00080           dxdy = (vfv.x - x) * inv_dy;
00081           const float Isz = vsv.z; // Z coord already inverted here
00082           const float Ifz = vfv.z;
00083           dIzdy = (Ifz - Isz) * inv_dy;
00084   
00085           // horizontal pixel correction
00086           float deltaX = dxdy *
00087             (vsv.y - (float (sy) - 0.5));
00088           x += deltaX;
00089   
00090           // apply sub-pixel accuracy factor
00091           float Factor;
00092           if (vfv.x != vsv.x)
00093             Factor = deltaX / (vfv.x - vsv.x);
00094           else
00095             Factor = 0;
00096   
00097           Iz = Lerp (Isz, Ifz, Factor);
00098           const float z = 1.0f/Iz;
00099   
00100             // Z coord already inverted here
00101           const float* Icsv = floats + sv*floatNum;
00102           const float* Icfv = floats + fv*floatNum;
00103           for (size_t f = 0; f < floatNum; f++)
00104           {
00105             const float fs = *Icsv++; 
00106             const float ff = *Icfv++; 
00107             Floats[f].Ic = Lerp (fs, ff, Factor);
00108             Floats[f].c = Floats[f].Ic * z;
00109             Floats[f].dIcdy = (ff - fs) * inv_dy;
