00118       NodeListType nodesToRemove;

00121       if (!childIt) return;

00126         //compare it to those we already have

00129         NodeListType::Iterator it = nodesToKeep.GetIterator ();
00130         while (it.HasNext ())
00131         {
00132           csRef<iDocumentNode> keepNode = it.Next ();
00133           if (keepNode->Equals (node))
00134           {
00135             keep = false;

00143         }
00144 
00145         if (keep)
00146         {
00147           nodesToKeep.Push (node);

00151           nodesToRemove.Push (node);
00152         }
00153       }
