00033   virtual int GetEntryCount () = 0;

00035   virtual bool GetLineNumber (int num, csString& str) = 0;

00037   void Print (FILE* f = stdout, bool Short = false)
00038   {
00039     for (int i = 0; i < GetEntryCount(); i++)
00040     {
00041       csString s;
00042       bool hasFunc = GetFunctionName (i, s);
00043       fprintf (f, "%s", hasFunc ? (const char*)s : "<unknown>");
00044       if (!Short && (GetLineNumber (i, s)))
00045         fprintf (f, " @%s", (const char*)s);
00046       if (!Short && (GetParameters (i, s)))
00047         fprintf (f, " (%s)", (const char*)s);
00048       fprintf (f, "\n");
00049     }
00050     fflush (f);
00051   }
00052 };
