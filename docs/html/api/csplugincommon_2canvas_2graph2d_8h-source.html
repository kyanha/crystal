00199   virtual int FindRGB (int r, int g, int b, int a = 255)
00200   {
00201     if (r < 0) r = 0; else if (r > 255) r = 255;
00202     if (g < 0) g = 0; else if (g > 255) g = 255;
00203     if (b < 0) b = 0; else if (b > 255) b = 255;
00204     if (a < 0) a = 0; else if (a > 255) a = 255;
00205     if (Depth == 8)
00206       return FindRGBPalette (r, g, b);
00207     return
00208       ((r >> (8 - pfmt.RedBits))   << pfmt.RedShift) |
00209       ((g >> (8 - pfmt.GreenBits)) << pfmt.GreenShift) |
00210       ((b >> (8 - pfmt.BlueBits))  << pfmt.BlueShift) |
00211       ((255 - a) << 24);
00212     /* Alpha is "inverted" so '-1' can be decomposed to a 
00213        transparent color. (But alpha not be inverted, '-1'
00214        would be "opaque white". However, -1 is the color
00215        index for "transparent text background". */
00216   }
