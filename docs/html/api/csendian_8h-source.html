00155 /*
00156     To be able to painlessly transfer files betwen platforms, we should
00157     avoid using native floating-point format. Here are a couple of routines
00158     that are guaranteed to work on all platforms.
00159 
00160     The floating point is converted to a fixed 1.7.25 bits format
00161     (one bit sign, 7 bits exponent, 25 bits mantissa) and back,
00162     so that we can binary store floating-point number without
00163     cross-platform problems. If you wonder why 1+7+25 = 33 while we
00164     only have 32 bits, we'll ommit the most significant bit of mantissa
00165     since it is always 1 (we use normalized numbers). This increases the
00166     precision twice.
00167 
00168     For double, we use one bit sign, 15 bits exponent, 49 bits mantissa.
00169 */
00170 
00175 static inline int32 csFloatToLong (float f)
00176 {
00177   int exp;
00178   int32 mant = csQroundSure (frexp (f, &exp) * 0x1000000);
00179   int32 sign = mant & 0x80000000;
00180   if (mant < 0) mant = -mant;
00181   if (exp > 63) exp = 63; else if (exp < -64) exp = -64;
00182   return sign | ((exp & 0x7f) << 24) | (mant & 0xffffff);
00183 }

00186 static inline float csLongToFloat (int32 l)
00187 {
00188   int exp = (l >> 24) & 0x7f;
00189   if (exp & 0x40) exp = exp | ~0x7f;
00190   float mant = float (l & 0x00ffffff) / 0x1000000;
00191   if (l & 0x80000000) mant = -mant;
00192   return (float) ldexp (mant, exp);
00193 }

00195 /* Implementation note: csDoubleToLongLong() and csLongLongToDouble()
00196  *
00197  * We avoid use of CONST_INT64() because 64-bit constants are illegal with g++
00198  * under -ansi -pedantic, and we want this header to be useful to external
00199  * projects which use -ansi -pedantic.  Instead, we use bit shifts, such as (1
00200  * << 59), and construct `mask' manually.
00201  */
00202 
00204 static inline int64 csDoubleToLongLong (double d)
00205 {
00206   int exp;
00207   int64 mant = (int64) (frexp (d, &exp) * ((int64)1 << 48));
00208   int64 sign = mant & ((int64)1 << 59);
00209   if (mant < 0) mant = -mant;
00210   if (exp > 32767) exp = 32767; else if (exp < -32768) exp = -32768;
00211   int64 const mask = ((uint64)0xffff << 32) | (uint64)0xffffffff;
00212   return sign | ((int64 (exp) & 0x7fff) << 48) | (mant & mask);
00213 }

00216 static inline double csLongLongToDouble (int64 i)
00217 {
00218   int exp = (i >> 48) & 0x7fff;
00219   if (exp & 0x4000) exp = exp | ~0x7fff;
00220   int64 const mask = ((uint64)0xffff << 32) | (uint64)0xffffffff;
00221   double mant = double (i & mask) / ((int64)1 << 48);
00222   if (i & ((int64)1 << 59)) mant = -mant;
00223   return ldexp (mant, exp);
00224 }

00235 static inline short csFloatToShort (float f)
00236 {
00237   int exp;
00238   long mant = csQroundSure (frexp (f, &exp) * 0x1000);
00239   long sign = mant & 0x8000;
00240   if (mant < 0) mant = -mant;
00241   if (exp > 7) mant = 0x7ff, exp = 7; else if (exp < -8) mant = 0, exp = -8;
00242   return short(sign | ((exp & 0xf) << 11) | (mant & 0x7ff));
00243 }

00246 static inline float csShortToFloat (short s)
00247 {
00248   int exp = (s >> 11) & 0xf;
00249   if (exp & 0x8) exp = exp | ~0xf;
00250   float mant = float ((s & 0x07ff) | 0x0800) / 0x1000;
00251   if (s & 0x8000) mant = -mant;
00252   return (float) ldexp (mant, exp);
