00337 inline void csReporterHelper::ReportV(iObjectRegistry* reg, int severity,
00338   char const* msgId, char const* description, va_list args)
00339 {
00340   csRef<iReporter> reporter;
00341   if (reg && (reporter = CS_QUERY_REGISTRY (reg, iReporter)))
00342     reporter->ReportV (severity, msgId, description, args);
00343   else
00344   {
00345     /*
00346       @@@ The csStrNCaseCmp()s are there because sometimes reported messages
00347       start with "Warning", and a "Warning: Warning" output looks rather
00348       crappy. The correct fix is obviously to remove "Warning" prefixes
00349       when the reporter is used.
00350      */
00351     switch (severity)
00352     {
00353       case CS_REPORTER_SEVERITY_BUG:
00354         csPrintf(CS_ANSI_BK CS_ANSI_FM CS_ANSI_FI "BUG: " CS_ANSI_RST);
00355         break;
00356       case CS_REPORTER_SEVERITY_ERROR:
00357         if (csStrNCaseCmp (description, "error", 5) != 0)
00358           csPrintf(CS_ANSI_BK CS_ANSI_FR CS_ANSI_FI "ERROR: " CS_ANSI_RST);
00359         break;
00360       case CS_REPORTER_SEVERITY_WARNING:
00361         if (csStrNCaseCmp (description, "warning", 7) != 0)
00362           csPrintf(CS_ANSI_BK CS_ANSI_FY CS_ANSI_FI "WARNING: " CS_ANSI_RST);
00363         break;
00364       case CS_REPORTER_SEVERITY_NOTIFY:
00365         csPrintf ("NOTIFY: ");
00366         break;
00367       case CS_REPORTER_SEVERITY_DEBUG:
00368         csPrintf(CS_ANSI_BK CS_ANSI_FW CS_ANSI_FI "DEBUG: " CS_ANSI_RST);
00369         break;
00370     }
00371     csPrintfV(description, args);
00372     csPrintf("\n");
00373   }
00374 }
