00207 // Platforms with compilers which understand the new C++ keyword `typename'
00208 // should define CS_HAVE_CXX_KEYWORD_TYPENAME. For other compilers, we fake up
00209 // a `typename' keyword which can be used to declare template arguments, such
00210 // as:
00211 //
00212 //   template<typename T> class A {...};
00213 //
00214 // Furthermore, we fake up a synthesized `typename_qualifier' keyword which
00215 // should be used to qualify types within a template declaration rather than
00216 // using `typename'. Usage example:
00217 //
00218 // template<typename T> struct A {
00219 //   typedef int B;
00220 //   typename_qualifier C::B var;
00221 //   typename_qualifier T::Functor get_functor() const;
00222 // };
00223 #if !defined(CS_HAVE_CXX_KEYWORD_TYPENAME)
00224   #define typename class
00225   #define typename_qualifier
00226 #else
00227   #define typename_qualifier typename
00228 #endif
