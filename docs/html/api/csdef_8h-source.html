00187 // Platforms with compilers which understand the new C++ keyword `typename'
00188 // should define CS_HAVE_CXX_KEYWORD_TYPENAME. For other compilers, we fake up
00189 // a `typename' keyword which can be used to declare template arguments, such
00190 // as:
00191 //
00192 //   template<typename T> class A {...};
00193 //
00194 // Furthermore, we fake up a synthesized `typename_qualifier' keyword which
00195 // should be used to qualify types within a template declaration rather than
00196 // using `typename'. Usage example:
00197 //
00198 // template<typename T> struct A {
00199 //   typedef int B;
00200 //   typename_qualifier C::B var;
00201 //   typename_qualifier T::Functor get_functor() const;
00202 // };
00203 #if !defined(CS_HAVE_CXX_KEYWORD_TYPENAME)
00204   #define typename class
00205   #define typename_qualifier
00206 #else
00207   #define typename_qualifier typename
00208 #endif
