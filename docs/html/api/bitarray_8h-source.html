00579   static int Compare (BitArray const& key1, BitArray const& key2)
00580   {
00581     csBitArrayStorageType const* p0 = key1.GetStore();
00582     csBitArrayStorageType const* p1 = key2.GetStore();
00583     size_t compareNum = MIN (key1.mLength, key2.mLength);
00584     size_t i = 0;
00585     for (; i < compareNum; i++)
00586       if (p0[i] != p1[i])
00587         return (int)p0[i] - (int)p1[i];
00588     if (key1.mLength > key2.mLength)
00589     {
00590       for (; i < key1.mLength; i++)
00591         if (p0[i] != 0)
00592           return (int)p0[i];
00593     }
00594     else
00595     {
00596       for (; i < key2.mLength; i++)
00597         if (p1[i] != 0)
00598           return -((int)p1[i]);
00599     }
00600     return 0;
00601   }
