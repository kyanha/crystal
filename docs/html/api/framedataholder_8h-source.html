00044   T& GetUnusedData (bool& created, uint frameNumber)
00045   {
00046     created = false;
00047     if ((data.Length() == 0) || (data[lastData].lastFrame == frameNumber))
00048     {
00049       lastData = (size_t)-1;
00050       //check the list
00051       size_t i;
00052       for(i = 0; i < data.Length (); i++)
00053       {
00054         if (data[i].lastFrame != frameNumber)
00055         {
00056           lastData = i;
00057           break;
00058         }
00059       }
00060       if (lastData == (size_t)-1)
00061       {
00062         data.GetExtend (lastData = data.Length ());
00063         created = true;
00064         nextShrink = frameNumber + 1;
00065       }
00066     }
00067   
00068     // Conserve some memory
00069     if (!created && (frameNumber <= nextShrink))
00070     {
00071       data.ShrinkBestFit ();
00072     }
00073   
00074     FrameData& frameData = data[lastData];
00075     frameData.lastFrame = frameNumber;
00076     return frameData.data;
00077   }
00078   
00079   void Clear ()
00080   {
00081     data.DeleteAll();
00082   }
00083 };
