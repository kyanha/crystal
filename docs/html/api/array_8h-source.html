00599   size_t Push (T const& what)
00600   {
00601     if (((&what >= root) && (&what < root + Length())) &&
00602       (capacity < count + 1))
00603     {
00604       /*
00605         Special case: An element from this very array is pushed, and a

00612       ElementHandler::Construct (root + count - 1, root[whatIndex]);
00613     }
00614     else
00615     {
00616       SetLengthUnsafe (count + 1);
00617       ElementHandler::Construct (root + count - 1, what);
00618     }
