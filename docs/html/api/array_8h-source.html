00611   size_t Push (T const& what)
00612   {
00613     if (((&what >= root) && (&what < root + Length())) &&
00614       (capacity < count + 1))
00615     {
00616       /*
00617         Special case: An element from this very array is pushed, and a
00618         reallocation is needed. This could cause the passed ref to the
00619         element to be pushed to be read from deleted memory. Work
00620         around this.
00621        */
00622       size_t whatIndex = &what - root;
00623       SetLengthUnsafe (count + 1);
00624       ElementHandler::Construct (root + count - 1, root[whatIndex]);
00625     }
00626     else
00627     {
00628       SetLengthUnsafe (count + 1);
00629       ElementHandler::Construct (root + count - 1, what);
00630     }
00631     return count - 1;
00632   }
