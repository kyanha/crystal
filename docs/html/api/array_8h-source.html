00604   size_t Push (T const& what)
00605   {
00606     if (((&what >= root) && (&what < root + Length())) &&
00607       (capacity < count + 1))

00609       /*

00620     {
00621       SetLengthUnsafe (count + 1);
00622       ElementHandler::Construct (root + count - 1, what);
00623     }
00624     return count - 1;
