00034   T PopTop ()
00035   {
00036     CS_ASSERT ((a1.Length() > 0) || (a2.Length() > 0));
00037     if (a2.Length() == 0)
00038     {
00039       size_t n = a1.Length();
00040       while (n-- > 0)
00041       {
00042         a2.Push (a1[n]);
00043       }
00044       a1.Empty();
00045     }
00046     return a2.Pop();
00047   }
00048 
00049   void Push (T const& what)
00050   {
00051     a1.Push (what);
00052   }
00053 
00054   size_t Length()
00055   {
00056     return a1.Length() + a2.Length();
00057   }
00058 
00059   bool Delete (T const& what)
00060   {
00061     return (a1.Delete (what) || a2.Delete (what));
00062   }
00063
