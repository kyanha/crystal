00909 class csStringFast : public csStringBase
00910 {
00911 protected:
00913   char minibuff[LEN];
00922   size_t miniused;
00923 
00924   virtual void SetCapacityInternal (size_t NewSize, bool soft)
00925   {
00926     if (Data != 0) // If dynamic buffer already allocated, just re-use it.
00927       csStringBase::SetCapacityInternal(NewSize, soft);
00928     else
00929     {
00930       NewSize++; // Plus one for implicit null byte.
00931       if (NewSize <= LEN)
00932         miniused = NewSize;
00933       else
00934       {
00935         CS_ASSERT(MaxSize == 0);
00936         if (soft)
00937           NewSize = ComputeNewSize (NewSize);
00938         Data = new char[NewSize];
00939         MaxSize = NewSize;
00940         if (Size == 0)
00941           Data[0] = '\0';
00942         else
00943           memcpy(Data, minibuff, Size + 1);
00944       }
00945     }
00946   }
00947

00949   { return (miniused == 0 && Data == 0 ? 0 : (Data != 0 ? Data : minibuff)); }

01000   virtual void ShrinkBestFit ()
01001   {
01002     if (Size == 0)
01003     {
01004       csStringBase::ShrinkBestFit();
01005       miniused = 0;
01006     }
01007     else
01008     {
01009       size_t needed = Size + 1;
01010       if (needed > LEN)
01011         csStringBase::ShrinkBestFit();
01012       else
01013       {
01014         miniused = needed;
01015         if (Data != 0)

01017           memcpy(minibuff, Data, needed); // Includes implicit null byte.
01018           csStringBase::Free();
