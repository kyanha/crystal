00916   virtual void SetCapacityInternal (size_t NewSize, bool soft)
00917   {
00918     if (Data != 0) // If dynamic buffer already allocated, just re-use it.
00919       csStringBase::SetCapacityInternal(NewSize, soft);
00920     else
00921     {
00922       NewSize++; // Plus one for implicit null byte.
00923       if (NewSize <= LEN)
00924         miniused = NewSize;
00925       else
00926       {
00927         CS_ASSERT(MaxSize == 0);
00928         if (soft)
00929           NewSize = ComputeNewSize (NewSize);
00930         Data = new char[NewSize];
00931         MaxSize = NewSize;
00932         if (Size == 0)
00933           Data[0] = '\0';
00934         else
00935           memcpy(Data, minibuff, Size + 1);
00936       }
00937     }
00938   }
