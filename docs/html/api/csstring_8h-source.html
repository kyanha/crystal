00964   virtual void SetCapacityInternal (size_t NewSize, bool soft)
00965   {
00966     if (Data != 0) // If dynamic buffer already allocated, just re-use it.
00967       csStringBase::SetCapacityInternal(NewSize, soft);
00968     else
00969     {
00970       NewSize++; // Plus one for implicit null byte.
00971       if (NewSize <= LEN)
00972         miniused = NewSize;
00973       else
00974       {
00975         CS_ASSERT(MaxSize == 0);
00976         if (soft)
00977           NewSize = ComputeNewSize (NewSize);
00978         Data = new char[NewSize];
00979         MaxSize = NewSize;
00980         if (Size == 0)
00981           Data[0] = '\0';
00982         else
00983           memcpy(Data, minibuff, Size + 1);
00984       }
00985     }
00986   }

00989   { return (miniused == 0 && Data == 0 ? 0 : (Data != 0 ? Data : minibuff)); }
