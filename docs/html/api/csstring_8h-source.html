00929   virtual void SetCapacityInternal (size_t NewSize, bool soft)
00930   {
00931     if (Data != 0) // If dynamic buffer already allocated, just re-use it.
00932       csStringBase::SetCapacityInternal(NewSize, soft);
00933     else
00934     {
00935       NewSize++; // Plus one for implicit null byte.
00936       if (NewSize <= LEN)
00937         miniused = NewSize;
00938       else
00939       {
00940         CS_ASSERT(MaxSize == 0);
00941         if (soft)
00942           NewSize = ComputeNewSize (NewSize);
00943         Data = new char[NewSize];
00944         MaxSize = NewSize;
00945         if (Size == 0)
00946           Data[0] = '\0';
00947         else
00948           memcpy(Data, minibuff, Size + 1);
00949       }
00950     }
00951   }

00954   { return (miniused == 0 && Data == 0 ? 0 : (Data != 0 ? Data : minibuff)); }
