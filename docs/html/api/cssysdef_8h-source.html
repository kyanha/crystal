00117 static inline bool isdir (const char *path, struct dirent *de)
00118 {
00119   int pathlen = strlen (path);
00120   char* fullname = new char[pathlen + 2 + strlen (de->d_name)];
00121   memcpy (fullname, path, pathlen + 1);
00122   if ((pathlen) && (fullname[pathlen-1] != CS_PATH_SEPARATOR))
00123   {
00124     fullname[pathlen++] = CS_PATH_SEPARATOR;
00125     fullname[pathlen] = 0;
00126   }
00127   strcat (&fullname [pathlen], de->d_name);
00128   struct stat st;
00129   stat (fullname, &st);
00130   delete[] fullname;
00131   return ((st.st_mode & S_IFMT) == S_IFDIR);
