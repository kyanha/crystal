00098 static inline bool isdir (const char *path, struct dirent *de)
00099 {
00100   int pathlen = strlen (path);
00101   char* fullname = new char[pathlen + 2 + strlen (de->d_name)];
00102   memcpy (fullname, path, pathlen + 1);
00103   if ((pathlen) && (fullname[pathlen-1] != CS_PATH_SEPARATOR))
00104   {
00105     fullname[pathlen++] = CS_PATH_SEPARATOR;
00106     fullname[pathlen] = 0;
00107   }
00108   strcat (&fullname [pathlen], de->d_name);
00109   struct stat st;
00110   stat (fullname, &st);
00111   delete[] fullname;
00112   return ((st.st_mode & S_IFMT) == S_IFDIR);
00113 }
