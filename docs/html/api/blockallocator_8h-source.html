00095   size_t FindBlock (void* m)
00096   {
00097     size_t i;
00098     for (i = 0 ; i < blocks.Length () ; i++)
00099     {
00100       csBlock* b = &blocks[i];
00101       if (b->memory <= m)
00102       {
00103         char* eb = ((char*)b->memory) + blocksize;
00104         if (((char*)m) < eb)
00105           return i;
00106       }
00107     }
00108     return (size_t)-1;

00200   {
00201     CS_ASSERT (blocks.Length() != 0);
00202     size_t i = blocks.Length () - 1;
00203     while (i > firstfreeblock)
00204     {
00205       if (blocks[i].firstfree == (csFreeList*)blocks[i].memory

00218   T* Alloc ()
00219   {
00220     CS_ASSERT (blocks.Length() != 0);
00221 
00222     // This routine makes sure there is ALWAYS free space available.
00223     csBlock& freebl = blocks[firstfreeblock];
00224     void* ptr = (void*)(freebl.firstfree);
00225 
00226     if (freebl.firstfree->numfree >= 2)
00227     {
00228       // Still room in this block after allocation.
00229       csFreeList* nf = (csFreeList*)(((char*)ptr)+elsize);
00230       nf->next = freebl.firstfree->next;
00231       nf->numfree = freebl.firstfree->numfree-1;
00232       freebl.firstfree = nf;

00234     else
00235     {
00236       // We need a new block later.
00237       freebl.firstfree = freebl.firstfree->next;
00238       if (!freebl.firstfree)

00240         // This block has no more free space. We need a new one.

00251   void Free (T* el)
00252   {
00253     if (!el) return;
00254 
00255     size_t idx = FindBlock ((void*)el);
00256     CS_ASSERT_MSG (
00257       "csBlockAllocator<>::Free() called with invalid element address",
00258       idx != (size_t)-1);
00259 
00260     el->~T();
00261 
00262 #ifdef CS_BLOCKALLOC_DEBUG
00263     memset (el, 0xfb, elsize);
00264 #endif
00265 
00266     // If the index is lower than the index of the first free block
00267     // then we update the firstfreeblock index.
00268     if (idx < firstfreeblock)
00269       firstfreeblock = idx;
00270 
00271     csBlock& bl = blocks[idx];
00272     if (bl.firstfree == 0)
00273     {
00274       // Block has no free items so we create the first free item
00275       // here.
00276       bl.firstfree = (csFreeList*)el;
00277       bl.firstfree->next = 0;
00278       bl.firstfree->numfree = 1;
00279     }
00280     else
00281     {
00282       csFreeList* p_el = (csFreeList*)el;
00283 
00284       if (p_el < bl.firstfree)
00285       {
00286         // New free element is before the current free element.
00287         if (((char*)bl.firstfree) - ((char*)p_el) == (int)elsize)
00288         {
00289           // New free element is directly before the current free
00290           // element.
00291           p_el->next = bl.firstfree->next;
00292           p_el->numfree = bl.firstfree->numfree+1;
00293         }
00294         else
00295         {
00296           // There is a gap.
00297           p_el->next = bl.firstfree;
00298           p_el->numfree = 1;
00299         }
00300 
00301         bl.firstfree = p_el;
00302       }
00303       else
00304       {
00305         // New free element is after the current free element.
00306         // First we find the two free blocks that enclose the new
00307         // free element.
00308         csFreeList* fl_before = bl.firstfree;
00309         csFreeList* fl_after = bl.firstfree->next;
00310         while (fl_after != 0 && fl_after < p_el)
00311         {
00312           fl_before = fl_after;
00313           fl_after = fl_after->next;
00314         }
00315 
00316         // 'fl_before_end' points to the memory right after the free block
00317         // that is directly before the new free element. We use
00318         // 'fl_before_end' later.
00319         char* fl_before_end = ((char*)fl_before) + fl_before->numfree * elsize;
00320 
00321         if (!fl_after)
00322         {
00323           // The new free element is after the last free block. First check
00324           // if the new free element directly follows that free block.
00325           // If so we can extend that.
00326           if (fl_before_end == (char*)p_el)
00327           {
00328             // Yes, extend.
00329             ++(fl_before->numfree);
00330           }
00331           else
00332           {
00333             // No, we have to create a new free block.
00334             p_el->next = 0;
00335             p_el->numfree = 1;
00336             fl_before->next = p_el;
00337           }
00338         }
00339         else
00340         {
00341           // We have a block before and after the new free block.
00342           // First we check if the new free item exactly between
00343           // fl_before and fl_after.
00344           if (fl_before_end == (char*)p_el
00345                 && ((char*)p_el) + elsize == (char*)fl_after)
00346           {
00347             // Perfect fit, merge fl_before and fl_after.
00348             fl_before->next = fl_after->next;
00349             fl_before->numfree = fl_before->numfree + 1 + fl_after->numfree;
00350           }
00351           else if (fl_before_end == (char*)p_el)
00352           {
00353             // New free item fits directly after fl_before.
00354             ++(fl_before->numfree);
00355           }
00356           else if (((char*)p_el) + elsize == (char*)fl_after)
00357           {
00358             // New free item fits directly before fl_after.
00359             fl_before->next = p_el;
00360             p_el->next = fl_after->next;
00361             p_el->numfree = fl_after->numfree+1;
00362           }
00363           else
00364           {
00365             // We need a new free block.
00366             fl_before->next = p_el;
00367             p_el->next = fl_after;
00368             p_el->numfree = 1;
00369           }
00370         }
00371       }
00372     }
00373   }
